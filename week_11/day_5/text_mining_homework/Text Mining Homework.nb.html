custom_stop_words <- bind_rows(tibble(word = c("miss"),  
                                      lexicon = c("custom")), 
                               stop_words)

pride_book <- tibble(
  text = prideprejudice,
    # treat each row as sentence
    sentence = 1:length(prideprejudice)
) %>% 
  unnest_tokens(word, text) %>% 
  anti_join(custom_stop_words)
